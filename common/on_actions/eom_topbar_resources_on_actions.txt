on_actions = {

	on_startup = {
		effect = {

			#log mod active
			set_global_flag = eom_TBR_active
			log = "EOM Topbar Resources active, updated for 1.19.*"

			every_country = {
				limit = {
					is_ai = no
				}
				set_country_flag = EOM_tbr_enabled
				EOM_TBR_update_gui = yes
			}
		}
	}

	on_daily = {
		effect = {
			if = {
				limit = {
					is_ai = no
				}
				EOM_TBR_update_gui = yes
			}
		}
	}
}
