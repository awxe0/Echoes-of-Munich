# Allows you to create your own dynamic keys
# to be called in localization.
# defined_text -> this is it, we're defining the text
# text -> a discrete entry that can be picked to display in loc.
# trigger -> determines if a text entry will be picked or not.
# (The triggers need to be valid for the scope the key is called in
# (eg Root or From.From).)
# localization_key -> points to the localization key
# that'll be used if trigger passes

defined_text = {
	name = eom_topbar_resource_toggle_button_text
	text = {
		trigger = {
			ROOT = { has_country_flag = OL_tbr_enabled }
		}
		localization_key = eom_topbar_resource_toggle_button_text_enabled
	}
	text = {
		trigger = {
			ROOT = { has_country_flag = OL_tbr_deficit }
		}
		localization_key = eom_topbar_resource_toggle_button_text_deficit
	}
	text = {
		trigger = {
			ROOT = { has_country_flag = OL_tbr_disabled }
		}
		localization_key = eom_topbar_resource_toggle_button_text_disabled
	}
	text = {
		localization_key = eom_topbar_resource_toggle_button_text_enabled
	}
}

defined_text = {
	name = OL_tbr_text_aluminium
	text = {
		trigger = {
			ROOT = {
				check_variable = { resource@aluminium < 0 }
			}
		}
		localization_key = OL_tbr_text_aluminium_red
	}
	text = {
		trigger = {
			ROOT = {
				check_variable = { resource_imported@aluminium > 0 }
			}
		}
		localization_key = OL_tbr_text_aluminium_yellow
	}
	text = {
	    trigger = {
			ROOT = {
				check_variable = { resource@aluminium > 0 }
			}
		}
		localization_key = OL_tbr_text_aluminium_green
	}
	text = {
		localization_key = OL_tbr_text_aluminium_black
	}
}
